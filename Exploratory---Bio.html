Change over time

# country mean values by year
avg.country = data.nomiss %>%
  group_by(Year) %>%
  summarise(avgobes = mean(obesity),
            avginact = mean(inactive),
            avgdiab = mean(diabetes))

# can we scale obesity & inactivity to plot with diabetes?
avg.avg.country = colMeans(avg.country)
mean(avg.avg.country[2:3]) / avg.avg.country[4]
##  avgdiab 
## 2.562472
# scale by 2.5

# convert to long format and scale
avg.country.long = rbind(
  avg.country %>% 
    mutate(scaleinact = avginact / 2.5) %>% 
    select(Year, USMean = scaleinact) %>% 
    mutate(Variable = "Inactivity"),
  avg.country %>% 
    select(Year, USMean = avgdiab) %>% 
    mutate(Variable = "Diabetes"),
  avg.country %>% 
    mutate(scaleobes = avgobes / 2.5) %>% 
    select(Year, USMean = scaleobes) %>% 
    mutate(Variable = "Obesity")
)

# 3 variables vs Year
ggplot(data = avg.country.long,
       aes(x = Year,
           y = USMean,
           color = Variable)) +
  geom_point() +
  geom_line() +
  scale_x_continuous(breaks = seq(2004, 2019, 5)) +
  scale_y_continuous(name = "Diabetes (%)",
                     sec.axis = sec_axis(trans = ~.*2.5, 
                                         name = "Inactivity or Obesity (%)",
                                         breaks = seq(18, 29, 3))) +
  labs(color = "") +
  theme_bw() +
  theme(panel.grid.major = element_blank(),
        panel.grid.minor = element_blank(),
        axis.ticks.length = unit(0.1, "inch"))

All three variables generally increase over time. Obesity and Inactivity have more sporadic but closely matching patterns, and Diabetes climbs more steadily.

# state-wide mean over time
avg.state = data.nomiss %>%
  group_by(State, Year) %>%
  summarise(Counties = n(),
            avginact = mean(inactive),
            avgdiab = mean(diabetes),
            avgobes = mean(obesity))

# Diabetes vs Year
ggplot(data = avg.state,
       aes(x = Year,
           y = avgdiab,
           group = State)) +
  geom_point(aes(size = Counties)) +
  geom_line() +
  scale_x_continuous(breaks = seq(2004, 2019, 5)) +
  scale_y_continuous(breaks = seq(5, 13, 2)) +
  scale_size_continuous(range = c(0.5, 3)) +
  labs(y = "Prevalence of Diabetes (%)") +
  theme_bw() +
  theme(panel.grid.major = element_blank(),
        panel.grid.minor = element_blank(),
        axis.ticks.length = unit(0.1, "inch"))

Diabetes generally increases in each state at similar rates, but different intercepts.
